上传SQL文件批量新增侦探故事，仅追加数据，不会清空原有内容

数据库两张表结构说明

-- 1.故事总表 story
story_id      自增主键
story_name    故事名称
story_desc    故事简介

-- 2.剧情页表 story_page
global_id        全局自增唯一主键（底层用）
story_id         绑定对应故事ID
local_page_id    本故事内独立页码
page_type        start开篇 / process过程 / ending结局
content          剧情文字，支持换行
options          JSON数组存储选项，jump_local_id跳转本故事页码
is_true_ending   1=真结局 0=普通结局

新增故事SQL模板（复制修改即可）

-- 【标准增量模板，必须带事务，出错自动全量回滚】
BEGIN TRANSACTION;

-- 第一步：新增故事，自动分配story_id
INSERT INTO story(story_name, story_desc)
VALUES ('你的故事名','故事简介');

-- 第二步：插入全部剧情页面
INSERT INTO story_page(story_id, local_page_id, page_type, content, options, is_true_ending) VALUES
(99,
1,
'start',
'开篇剧情文字',
'[{"text":"选项1","jump_local_id":2}]',
0),
(99,
9,
'ending',
'真结局文案',
'[]',
1);

-- 无报错才会保存入库
COMMIT;
