-- story表：存储故事基本信息
CREATE TABLE IF NOT EXISTS story (
    story_id INTEGER PRIMARY KEY AUTOINCREMENT,
    story_name TEXT NOT NULL,          -- 故事名称（必填）
    story_desc TEXT                    -- 故事简介（可选）
);

-- story_page表：存储剧情页内容
CREATE TABLE IF NOT EXISTS story_page (
    story_id INTEGER NOT NULL,
    local_page_id INTEGER NOT NULL,     -- 故事内页ID（必填，同一故事内唯一）
    content TEXT NOT NULL,              -- 页面正文（必填）
    options TEXT,                       -- 分支选项（JSON格式，例：[{"text":"选项1","jump_local_id":2}]）
    page_type TEXT CHECK(page_type IN ('start','process','ending')) NOT NULL, -- 页面类型
    is_true_ending BOOLEAN DEFAULT 0,  -- 是否真结局（1=真结局，0=假结局）
    pos_x INTEGER DEFAULT 50,           -- 思维导图X坐标（可选）
    pos_y INTEGER DEFAULT 50,           -- 思维导图Y坐标（可选）
    PRIMARY KEY (story_id, local_page_id),
    FOREIGN KEY (story_id) REFERENCES story(story_id) ON DELETE CASCADE
);
  • 导入会覆盖同ID的故事数据，请提前备份重要内容
  • SQL文件编码必须为UTF-8，否则会出现乱码
  • 支持批量导入多个故事，请确保`story_id`不重复
